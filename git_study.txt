4장. 커밋
git add .
git rm --cached main.py
git mv 파일이름 새이름
git commit -am '메시지'
git config --global core.editor 'vim이외 에디터 경로'
git commit --allow-empty-message -am ""
git commit --amend (방금 커밋메시지 수정)
git log --pretty=short (앞줄)
git show '커밋ID'
git diff
